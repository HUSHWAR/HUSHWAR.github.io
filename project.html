class button{
	
	private:
		SDL_Surface* button_image = NULL;
		SDL_Rect dest;
		int x;
		int y;
		int w;
		int h;
		
	public:
		buttonstatus state;
		button(int tx,int ty,int tw,int th,char * image){
			x = tx;
			y = ty;
			w = tw;
			h = th;
			LoadImage(image);
		}
		
		void check(SDL_Event* event){
		    int mx, my;
		    SDL_GetMouseState(&mx, &my);
		    switch(event->type){
		        case SDL_MOUSEBUTTONDOWN: if(mx >= x && mx <= x + w && my >= y && my <= y + h){state = PRESSED;}break;
		        case SDL_MOUSEMOTION:
		            if(mx >= x && mx <= x + w && my >= y && my <= y + h){
		                state = MOUSEENTER;
		            }
		            else
		                state = MOUSEOUT;
		            break;
		        case SDL_MOUSEBUTTONUP: if(mx >= x && mx <= x + w && my >= y && my <= y + h){ state = MOUSEENTER;}break;
		    }
		}
			
		void Draw() {
			dest.x = x;
		  	dest.y = y;
			dest.w = w;
		  	dest.h = h;
		    
		  	SDL_BlitScaled(button_image,
		                   NULL,
		                   screen,
		                   &dest);
		}	
	
		void LoadImage(char * image_name){
			
			if(image_name!="invisible"){
			SDL_Surface* tsurface;
			
			tsurface = IMG_Load(image_name);
			if(tsurface == NULL){
				printf("Can't load image %s\n",image_name);
				button_image = NULL;
			}
			//printf("Button loaded %s\n",image_name);
			button_image = tsurface;
			}else{
			button_image = NULL;
			}
		}
};



int VInit() {

    win = SDL_CreateWindow("Project 1", SDL_WINDOWPOS_UNDEFINED, SDL_WINDOWPOS_UNDEFINED, SCREEN_WIDTH, SCREEN_HEIGHT, SDL_WINDOW_SHOWN);
    if (win == NULL) {return 123;}

    screen = SDL_GetWindowSurface(win);

    return 0;
}

SDL_Surface* LoadFont(char * BitmapFont){
  SDL_Surface* temp;
  
  temp = IMG_Load(BitmapFont);
  if(temp == NULL){
    printf("Can't load image %s\n",BitmapFont);
  return NULL;
  }
  
  return temp;
}

void DrawText(text_type tt,SDL_Surface* sFont, int x, int y, char * str,int size) {
  SDL_Rect src, dest;
  
  switch (tt){
    
  case TYPE_WORD:  
  for (int i = 0; i < strlen(str); i++) {
     if (str[i] >= 'À' && str[i] <= 'ß') {
	src.x = (str[i] - 'À') * 10;
	src.y = 0;
  	//=======================
	//src.w = 32;
  	//src.h = 32;
    //=======================
    src.w = 10;
  	src.h = 12;

	dest.x = x + (i *size);
  	dest.y = y;
	dest.w = size;
  	dest.h = size;

  	SDL_BlitScaled(sFont,
                   &src,
                   screen,
                   &dest);
  	
    }}
	break;
	
case TYPE_NUMBERS:
	for (int i = 0; i < strlen(str); i++) {
     if (str[i] >= '1' && str[i] <= '9') {
	src.x = ((str[i] - '1')+1) * 10;
	src.y = 0;
  	
	src.w = 10;
  	src.h = 12;

	dest.x = x + (i *size);
  	dest.y = y;
	dest.w = size;
  	dest.h = size;

  	SDL_BlitScaled(sFont,
                   &src,
                   screen,
                   &dest);
  	
    }
	else
	{
    	switch(str[i])
    	{
    	case '=':
		src.x = 100;
    	src.y = 0;
    	break;
    	case '*':
		src.x = 110;
    	src.y = 0;
    	break;
		break;
    	case '+':
		src.x = 120;
    	src.y = 0;
    	break;
    	case '-':
		src.x = 130;
    	src.y = 0;
    	break;
		case '?':
		src.x = 140;
    	src.y = 0;
    	break;
    	case ' ':
		src.x = 150;
    	src.y = 0;
    	break;
    	
		}
		
		src.w = 10;
	  	src.h = 12;
		
		
		dest.x = x + (i *size);
	  	dest.y = y;
		dest.w = size;
	  	dest.h = size;
	
	  	SDL_BlitScaled(sFont,
	                   &src,
	                   screen,
	                   &dest);
	}
	}
	break;	
	}//Switch(tt)
}
